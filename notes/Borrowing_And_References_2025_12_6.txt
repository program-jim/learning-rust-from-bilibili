// Borrowing and References are powerful concepts.

// Understanding References
// References: Enable you to borrow values without taking ownership.
// Immutable references
// Mutable references
// Create references by add "&"

// Demonstration on one mutable reference or many immutable references
fn main() {
    let mut x = 5;
    let r = &mut x;
    *r += 1;
    *r -= 3;
    println!("r: {}", r);
    println!("x: {}", x);

    let mut account = BankAccount {
        owner : "Alice".to_string(),
        balance : 150.55,
    };

    // Immutable borrow to check the balance.
    account.check_balance();

    // Mutable borrow to withdraw money.
    account.withdraw(50.5);

    // Check the Balance again.
    account.check_balance();
}

// STRUCT:
// A data structure that allows you
// to group multiple fields
// together under one name.
// (also exists in C and C++)
struct BankAccount {
    owner: String,
    balance: f64,
}

impl BankAccount {
    fn withdraw(&mut self, amount: f64) {
        println!("Withdrawing {} from account owned by {}", amount, self.owner);
        self.balance -= amount;
    }

    fn check_balance(&self) {
        println!("Account owned by {} has a balance of {}", self.owner, self.balance);
    }
}
